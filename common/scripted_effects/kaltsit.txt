create_kaltist_official = {
    every_owned_leader = {
        limit = { has_leader_flag = kaltsit }
        kill_leader = { show_notification = no }
    }
    set_country_flag = owned_kaltsit
    create_leader = {
        class = official
        skill = 10
        name = kaltsit_name
        custom_catch_phrase = kaltsit_official_intro
        tier = leader_tier_legendary
        immortal = yes #不朽
        gender = female
        species = owner_main_species
        set_age = 1
        hide_age = yes
        event_leader = yes
        traits = {
            trait = leader_trait_kaltsit_official
        }
        effect = {
            set_leader_flag = kaltsit
            set_leader_flag = kaltsit_official
        }
    }
    last_created_leader = {
        change_leader_portrait = kaltsit_official
    }
}
